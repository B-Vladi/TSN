 18 Parser.prototype.onStart = function () {
 19 	var code;
 20 	this.current.code = '';
 21 
 22 	for (var nodeName in nodeAPI) {
 23 		if (nodeAPI.hasOwnProperty(nodeName) && nodeAPI[nodeName].hasOwnProperty('start')) {
 24 			code = nodeAPI[nodeName].start(this, TSN);
 25 
 26 			if (typeof code == 'string') {
 27 				this.current.code += code;
 28 			}
 29 		}
 30 	}
 31 };
 32 
 33 Parser.prototype.onEnd = function () {
 34 	var code;
 35 
 36 	for (var nodeName in nodeAPI) {
 37 		if (nodeAPI.hasOwnProperty(nodeName) && nodeAPI[nodeName].hasOwnProperty('end')) {
 38 			code = nodeAPI[nodeName].end(this, TSN);
 39 
 40 			if (typeof code == 'string') {
 41 				this.current.code += code;
 42 			}
 43 		}
 44 	}
 45 };

 47 Parser.prototype.onText = function (text, node) {
 48 	node.text += text.replace(/^<!\[CDATA\[([\s\S]*?)\]\]>$/i, '$1');
 49 	node.code += (this.inline === true ? ' + ' : '__text = ') + '"' + this.fixText(text) + '"';
 50 
 51 	this.inline = true;
 52 };
 53 
 54 Parser.prototype.onEntity = function (node) {
 55 	node.parent.code += (this.inline === true ? ' + ' : '__text = ') + node.name;
 56 	this.inline = true;
 57 };

 63 Parser.prototype.onOpen = function (node) {
 64 	if (nodeAPI.hasOwnProperty(node.name)) {
 65 		var API = nodeAPI[node.name];
 66 
 67 		node.template = API.template;
 68 		node.parse = API.parse;
 69 		node.inline = API.inline;
 70 		node.code = ';';
 71 		node.text = '';
 72 
 73 		if (node.isEmpty) {
 74 			var parseResult = typeof node.parse === 'function' ? node.parse(this, TSN) : true;
 75 
 76 			if (parseResult && parseResult.constructor === Error) {
 77 				this._error(parseResult.message, node);
 78 			} else {
 79 				node.parent.code += compileNode(node, this);
 80 			}
 81 		} else {
 82 			this.inline = false;
 83 		}
 84 	} else {
 85 		this._error(node.isEmpty ? 'Unknown empty tag.' : 'Unknown tag opening.', node);
 86 	}
 87 };
 88 
 89 Parser.prototype.onClose = function (node) {
 90 	if (nodeAPI.hasOwnProperty(this.current.name)) {
 91 		var parseResult = typeof this.current.parse === 'function' ? this.current.parse(this, TSN) : true;
 92 
 93 		if (parseResult && parseResult.constructor === Error) {
 94 			this._error(parseResult.message, this.current);
 95 		} else {
 96 			this.current.parent.code += compileNode(this.current, this);
 97 		}
 98 	} else {
 99 		this._error('Unknown tag closing.', node);
100 	}
101 };

103 Parser.prototype.fixText = function (text) {
104 	var tabSize, spaceSize;
105 
106 	if (this.depth) {
107 		tabSize = this.depth * (this.config.indent / this.config.tabSize);
108 		spaceSize = this.depth * this.config.indent;
109 
110 		text = text.replace(new RegExp('(?:(?:\\r\\n)|\\r|\\n)(?:[\\t]{' + tabSize + '}|[ ]{' + spaceSize + '})', 'g'), '\n');
111 	}
112 
113 	return text
114 		.replace(/\\/g, '\\\\')
115 		.replace(/("|')/g, '\\$1')
116 		.replace(/(?:\r\n)|\r|\n/g, '\\n')
117 		.replace(/\f/g, '\\f')
118 		.replace(/\u2028/g, '\\u2028')
119 		.replace(/\u2029/g, '\\u2029');
120 };

122 function compileNode (node, parser) {
123 	var code = node.template.replace(/\/\*(?:(!|@)([a-z\-_]+)?)\*\//gi, function (result, type, name) {
124 		switch (type) {
125 			case '!':
126 				switch (name) {
127 					case 'code':
128 						if (node.inline !== true) {
129 							node.code += ';' +
130 								'__output += __text;' +
131 								'__hasStream && __text !== "" && __stream.write(__text, "' + parser.config.encoding + '");' +
132 								'__text = "";';
133 						}
134 
135 						return node.code;
136 					case 'context':
137 						return node.attributes.hasOwnProperty('context') ? node.attributes.context : 'this';
138 				}
139 				break;
140 			case '@':
141 				return node.attributes[name];
142 		}
143 	});
144 
145 	if (node.inline === true) {
146 		code = parser.inline === true ? ' + ' + code : '__text = ' + code;
147 	} else {
148 		code = ';' +
149 			'__output += __text;' +
150 			'__hasStream && __text !== "" && __stream.write(__text, "' + parser.config.encoding + '");' +
151 			'__text = "";' + code;
152 	}
153 
154 	parser.inline = node.inline;
155 
156 	return code;
157 }

190 /**
191  * Компиляция шаблона.
192  * @param {string} source Исходный код шаблона.
193  * @param {object} [config] Объект конфигурации шаблона: {@link TSN.config}.
194  * @return {function} Скомпилированный шаблон.
195  */
196 TSN.compile = function (source, config) {
197 	config = new Config(config);
198 
199 	if (config.cache === true && config.hasOwnProperty('name') && TSN.cache.hasOwnProperty(config.name)) {
200 		return TSN.cache[config.name];
201 	}
202 
203 	var template = new Parser(source, config);
204 
205 	source = '' +
206 		'"use strict";' +
207 
208 		'var __name = "' + config.name.replace(/('|"|(?:\r\n)|\r|\n|\\)/g, "\\$1") + '";' +
209 		'var __output = "";' +
210 		'var __text = "";' +
211 		'var __hasStream = __stream !== void 0;' +
212 
213 		'try {' +
214 			template.root.code +
215 			';' +
216 			'__output += __text;' +
217 			'__hasStream && __text !== "" && __stream.write(__text, "' + config.encoding + '");' +
218 		'} catch (error) {' +
219 			'error.templateName = __name;' +
220 			'error.TypeError = "RenderError";' +
221 
222 			'TSN.emit("error", error);' +
223 		'}' +
224 
225 		'return __output;';
226 
227 	template = new Function('__stream', 'TSN', source);
228 
229 	template.call = call;
230 	template.apply = apply;
231 	template.source = source;
232 
233 	if (config.cache === true && typeof config.name === 'string' && config.name !== '') {
234 		template.cacheName = config.name;
235 		TSN.cache[config.name] = template;
236 	}
237 
238 	TSN.emit('compile', template);
239 
240 	return template;
241 };

243 /**
244  * Синхронная компиляция шаблона из файла.
245  * @param {string} path Путь к файлу шаблона относительно <i>TSN.config.templateRoot</i>.
246  * @param {object} [config] Объект конфигурации шаблона: {@link TSN.config}.
247  * @return {function} Скомпилированный шаблон.
248  */
249 TSN.compileFromFile = function (path, config) {
250 	config = new Config(config);
251 
252 	var fullPath = LIB.path.join(config.templateRoot, path);
253 
254 	if (config.cache === true && TSN.cache.hasOwnProperty(fullPath)) {
255 		return TSN.cache[fullPath];
256 	}
257 
258 	if (!config.hasOwnProperty('name')) {
259 		config.name = fullPath;
260 	}
261 
262 	config.path = LIB.path.dirname(fullPath);
263 	return TSN.compile(LIB.fileSystem.readFileSync(fullPath, config.encoding), config);
264 };

266 /**
267 * Асинхронная рекурсивная компиляция файлов.
268 * @param {RegExp} [pattern=/.* /] Ругелярное выражение, которому должно соответствовать имя файла для компиляции.
269 * @param {string} [path=TSN.config.templateRoot] Путь к дирректории, в которой необходимо компилировать файлы.
270 */
271 TSN.compileFromDir = function (pattern, path) {
272 	if (!(pattern instanceof RegExp)) {
273 		pattern = /.*/;
274 	}
275 
276 	if (typeof path !== 'string') {
277 		path = TSN.config.templateRoot;
278 	}
279 
280 	var directory = {
281 		state: 'start',
282 		path: path,
283 		dirsLength: 0
284 	};
285 
286 	LIB.fileSystem.stat(path, callback);
287 
288 	function callback (error, data) {
289 		if (error) {
290 			error.TypeError = 'CompileDirError';
291 			TSN.emit('error', error);
292 			return;
293 		}
294 
295 		switch (directory.state) {
296 			case 'start':
297 				if (data.isDirectory()) {
298 					directory.state = 'read';
299 
300 					LIB.fileSystem.readdir(directory.path, callback);
301 				} else {
302 					error = new Error('Path ' + directory.path + ' is not a directory.');
303 					error.TypeError = 'CompileDirError';
304 
305 					TSN.emit('error', error);
306 				}
307 				break;
308 			case 'read':
309 				directory.files = data;
310 				directory.state = 'status';
311 
312 				if (data.length) {
313 					directory.currentFile = data.shift();
314 
315 					LIB.fileSystem.stat(LIB.path.join(directory.path, directory.currentFile), callback);
316 				} else {
317 					callback(error, new LIB.fileSystem.Stats);
318 				}
319 				break;
320 			case 'status':
321 				var path = LIB.path.join(directory.path, directory.currentFile);
322 
323 				if (data.isFile()) {
324 					if (pattern.test(directory.currentFile)) {
325 						TSN.load(LIB.path.relative(directory.root || directory.path, path));
326 					}
327 				} else if (data.isDirectory()) {
328 					var child = TSN.compileDir(pattern, path);
329 
330 					child.root = directory.root || directory.path;
331 					child.parent = directory;
332 
333 					directory.dirsLength++;
334 				}
335 
336 				if (directory.files.length) {
337 					directory.currentFile = directory.files.shift();
338 					LIB.fileSystem.stat(LIB.path.join(directory.path, directory.currentFile), callback);
339 				} else {
340 					var dir = directory;
341 					directory.state = 'end';
342 
343 					while (!dir.dirsLength) {
344 						if (dir.parent) {
345 							dir = dir.parent;
346 							dir.dirsLength--;
347 						} else if (dir.state === 'end') {
348 							TSN.emit('compileDirEnd', dir.root || dir.path);
349 							break;
350 						}
351 					}
352 
353 				}
354 				break;
355 		}
356 	}
357 
358 	return directory;
359 };
